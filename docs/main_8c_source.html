155 void reshape(int w, int h) {
156  ClientRect* vp = &GameState.viewport;
157  ClientRect* ortho = &GameState.ortho;
158 
159  double ratio = (double)MAZE_SIZE / (MAZE_SIZE + HUD_HEIGHT);
160  double size = clamp(min(/* w */ h * ratio, /* h */ w / ratio),
161  0, min(w, h));
162 
163  vp->width = size;
164  vp->height = size / ratio;
165  vp->left = (w - vp->width) / 2;
166  vp->bottom = (h - vp->height) / 2;
167  vp->top = vp->bottom + vp->height;
168  vp->right = vp->left + vp->width;
169 
170  glViewport(vp->left, vp->bottom, vp->width, vp->height);
171  glMatrixMode(GL_PROJECTION);
172  glLoadIdentity();
173 
174  ortho->top = MAZE_SIZE + HUD_HEIGHT;
175  ortho->right = MAZE_SIZE;
176  ortho->bottom = 0;
177  ortho->left = 0;
178  ortho->width = ortho->right - ortho->left;
179  ortho->height = ortho->top - ortho->bottom;
180 
181  gluOrtho2D(ortho->left, ortho->right, ortho->bottom, ortho->top);
182 
183  glMatrixMode(GL_MODELVIEW);
184 }
