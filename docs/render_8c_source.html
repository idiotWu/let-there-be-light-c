36 static void renderTiles(void) {
37  // render one more tile
38  double r = GameState.visibleRadius + 1.0;
39 
40  size_t xMin = max(floor(GameState.player.x - r), 0);
41  size_t xMax = min(ceil(GameState.player.x + r), MAZE_SIZE - 1);
42  size_t yMin = max(floor(GameState.player.y - r), 0);
43  size_t yMax = min(ceil(GameState.player.y + r), MAZE_SIZE - 1);
44 
45  for (size_t y = yMin; y <= yMax; y++) {
46  for (size_t x = xMin; x <= xMax; x++) {
47  Tile tile = GameState.maze[y][x];
48 
49  // path
50  if (tile & TILE_OPEN) {
51  int row, col;
52  if (tile & TILE_SPOILED) {
53  row = FROZEN_PATH_ROW;
54  col = FROZEN_PATH_COL;
55  } else {
56  row = PATH_ROW;
57  col = PATH_COL;
58  }
59 
60  renderSprite(MISC_SPRITES, row, col, x, y, 1, 1);
61  }
62 
63  // other items
64  int row, col;
65 
66  if (tile & TILE_SHADOW) {
67  row = SHADOW_ROW;
68  col = SHADOW_COL;
69  } else if(tile & TILE_COIN) {
70  row = COIN_ROW;
71  col = COIN_COL;
72  } else if (tile & TILE_KERNEL) {
73  row = FIRE_ROW;
74  col = FIRE_COL;
75  } else {
76  continue;
77  }
78 
79  renderSprite(MISC_SPRITES, row, col, x, y, 1, 1);

138 static void renderEnemies(void) {
139  ListIterator it = createListIterator(GameState.enemies);
140 
141  while (!it.done) {
142  Node* node = it.next(&it);
143  Enemy* enemy = node->data;
144  double ox = enemy->x + 0.5;
145  double oy = enemy->y + 0.5;
146  double scale = min(1.0, (double)enemy->remainSteps / ENEMY_MAX_STEPS + 0.2);
147 
148  glPushMatrix();

150  glScaled(scale, scale, 1.0);
151  glTranslated(-ox, -oy, 0.0);
152 
153  renderCharacter(ENEMY_SPRITES, enemy->spriteState, enemy->x, enemy->y, enemy->direction);
154  glPopMatrix();
155  }
