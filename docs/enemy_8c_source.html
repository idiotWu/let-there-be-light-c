119 static vec2i getEnemyDelta(Enemy* enemy) {
120  int x = enemy->x;
121  int y = enemy->y;
122  int minSteps = GameState.stepsFromPlayer[y][x] - 1;
123 
124  vec2i delta = { 0, 0 };
125 
126  if (minSteps < 0) {
127  // in case player is moving
128  // -> move towards player
129  delta.x = sign(GameState.player.x - x);
130  delta.y = sign(GameState.player.y - y);
131 
132  if (!isPath(GameState.maze, x + delta.x, y)) {
133  delta.x = 0;
134  }
135 
136  if (!isPath(GameState.maze, x, y + delta.y)) {
137  delta.y = 0;
138  }
139 
140  return delta;
141  }
142 
143  if (GameState.stepsFromPlayer[y][x - 1] == minSteps) {
144  delta.x = -1;
145  } else if (GameState.stepsFromPlayer[y][x + 1] == minSteps) {
146  delta.x = 1;
147  } else if (GameState.stepsFromPlayer[y - 1][x] == minSteps) {
148  delta.y = -1;
149  } else if (GameState.stepsFromPlayer[y + 1][x] == minSteps) {
150  delta.y = 1;
151  }
152 
153  return delta;
154 }
