62 static void initItems(void) {
63  Tile (*tiles)[MAZE_SIZE] = GameState.maze;
64 
65  double itemDesity = getItemDensity();
66 
67  int openCount = 0;
68  GameState.remainItem = 0;
69 
70  for (int y = 0; y < MAZE_SIZE; y++) {
71  for (int x = 0; x < MAZE_SIZE; x++) {
72  if (tiles[y][x] & TILE_OPEN) {
73  vec2i pos = { x, y };
74  GameState.openTiles[openCount++] = pos;
75 
76  // clear items randomly
77  if (randomBetween(0, 1) >= itemDesity) {
78  clearBits(tiles[y][x], TILE_ITEM);
79  } else {
80  GameState.remainItem++;
81  }
82  } else if (y != MAZE_SIZE - 1 && tiles[y + 1][x] & TILE_OPEN) {
83  tiles[y][x] = TILE_SHADOW;
84  }
85  }
86  }
87 }

129 /**
130  * @brief 押している方向キーを読み取り，プレイヤーを移動させる
131  */
132 static void readKeyboard(void) {
133  Direction moveDir = getPressedKeys();
134 
135  if (moveDir == DIR_NONE) {
136  return;
137  }
138 
139  int x = GameState.player.x;
140  int y = GameState.player.y;
141 
142  vec2i delta = { 0, 0 };
143 
144  if (moveDir & DIR_LEFT) {
145  delta.x--;
146  }
147  if (moveDir & DIR_RIGHT) {
148  delta.x++;
149  }
150  if (moveDir & DIR_UP) {
151  delta.y++;
152  }
153  if (moveDir & DIR_DOWN) {
154  delta.y--;
155  }
156 
157  if (!isPath(GameState.maze, x + delta.x, y)) {
158  delta.x = 0;
159  }
160  if (!isPath(GameState.maze, x, y + delta.y)) {
161  delta.y = 0;
162  }
163 
164  if (delta.x || delta.y) {
165  GameState.player.idle = false;
166  movePlayer(delta.x, delta.y);
167  }
168 }

203  tried++;

205 
206  free(GameState.openTiles);
207 
208  GameState.paused = false;
209  GameState.pathLength = pathLength;
210  GameState.openTiles = malloc(pathLength * sizeof(*GameState.openTiles));
211  GameState.visibleRadius = getInitialVisibleRadius();
212 
213  initItems();

215  initEnemy();

217  bindGameKeyboardHandlers();

242 static void updateGame(void) {
243  if (GameState.paused) {
244  return;
245  }
246 
247  GameState.visibleRadius -= getVisibleRadiusRuducingRate();
248 
249  // game over
250  if (GameState.visibleRadius <= 0) {
251  return switchScene(gameOverScene);
252  }
253 
254  Player* player = &GameState.player;
255 
256  player->spoiled = (GameState.maze[(int)player->y][(int)player->x] & TILE_SPOILED);
257 
258  // move player when it's idle(not moving)
259  if (player->idle) {
260  updateItems();
261  readKeyboard();
262  }
263 
264  // activate sleeping enemies
265  activateEnemies();
266 
267  // stage clear
268  if (GameState.remainItem == 0) {
269  player->spoiled = false;
270  GameState.paused = true;
271  GameState.lastVisibleRadius = GameState.visibleRadius;
272 
273  // show maze
274  expandVision(MAZE_SIZE * 10, 3000);
275  // remove all enemies
276  destroyEnemy(true);
277  // clear spoiled tiles
278  fxFlood(player->x, player->y);
279 
280  delay(5000, gameClear, NULL);
281  }
282 }
