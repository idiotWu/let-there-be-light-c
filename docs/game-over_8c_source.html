79 /**
80  * @brief メッセージを画面の中央に出力する
81  *
82  * @param str メッセージ
83  */
84 static void showMessage(const char* str) {
85  size_t length = strlen(str);
86  double width = GameState.ortho.width;
87  double height = GameState.ortho.height;
88 
89  char* tmp = malloc(sizeof(char) * (length + 1));
90  strcpy(tmp, str);
91 
92  int randomIdx = randomInt(0, (int)length - 1);
93 
94  if (tmp[randomIdx] == ' ') {
95  randomIdx--;
96  }
97 
98  // replace one letter randomly
99  tmp[randomIdx] = RANDOM_CHAR;
100 
101  setTexParam(GL_MODULATE);
102  glColor3d(0.6, 0.2, 0.2);
103  renderText(tmp, (width - length * CAPTION_FONT_SIZE) / 2.0, (height - CAPTION_FONT_SIZE) / 2.0, CAPTION_FONT_SIZE);
104  restoreDefaultTexParam();
105 
106  free(tmp);
107 }
