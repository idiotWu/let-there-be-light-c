72  uint32_t dimension = 0;

119 static GLuint createTextureFrom(const char* filename) {
120  GLuint texName;

122

137  uint32_t height = readDimension(fp, 4);
138  size_t byteLength = width * height * 4;
139

142  fseek(fp, 8, SEEK_SET);

158 static GLuint initFogTexture() {
159  GLuint texName;
160 
161  GLubyte pixels[FOG_SIZE * FOG_SIZE * 4];
162  double radius = FOG_SIZE / 2.0;
163 
164  for (size_t i = 0; i < FOG_SIZE; i++) {
165  for (size_t j = 0; j < FOG_SIZE; j++) {
166  size_t offset = (i * FOG_SIZE + j) * 4;
167 
168  // rgb = (0, 0, 0)
169  pixels[offset] = pixels[offset + 1] = pixels[offset + 2] = 0;
170 
171  double distance = sqrt(pow(i - radius, 2) + pow(j - radius, 2));
172 
173  // alpha
174  pixels[offset + 3] = (GLubyte)(0xff * min(1.0, distance / radius));
175  }
176  }
177 
178  bindTexture(&texName, FOG_SIZE, FOG_SIZE, pixels);
179 
180  return texName;
181 }

198  double sw, double sh,
199  double dx, double dy,
200  double dw, double dh) {
